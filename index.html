Goodbye CoffeeScript,
hello JavaScript!

# install locally            
 $ npm install -g decaffeinate

# convert a single file      
 $ decaffeinate index.coffee  

# convert a whole project    
 $ decaffeinate src/          

# remove the .coffee files   
 $ find . -name '*.coffee' \  
> -exec rm {} \;             

# commit the result           
 $ git add -A                  
$ git commit
